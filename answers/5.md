# BPMN - Ví dụ Cụ thể và Code Example

## 1. Ví dụ thực tế: Quy trình xét duyệt đơn vay vốn

Chúng ta sẽ xây dựng một **Loan Approval Process** (Quy trình xét duyệt đơn vay vốn) hoàn chỉnh.

### Mô tả nghiệp vụ

```
1. Khách hàng gửi đơn vay vốn
2. Hệ thống kiểm tra thông tin khách hàng
3. Kiểm tra điểm tín dụng (credit score)
4. Nếu credit score >= 700 → Tự động duyệt
5. Nếu 500 <= credit score < 700 → Chuyển qua nhân viên xét duyệt
6. Nếu credit score < 500 → Từ chối
7. Gửi thông báo kết quả
```

## 3. BPMN 2.0 XML Code

Đây là file XML thực tế của BPMN diagram trên (file `.bpmn`):

```xml
<?xml version="1.0" encoding="UTF-8"?>
<bpmn:definitions xmlns:bpmn="http://www.omg.org/spec/BPMN/20100524/MODEL"
                  xmlns:bpmndi="http://www.omg.org/spec/BPMN/20100524/DI"
                  id="LoanApprovalProcess">

  <bpmn:process id="loanApproval" name="Loan Approval Process" isExecutable="true">

    <!-- Start Event -->
    <bpmn:startEvent id="StartEvent_1" name="Loan Application Received">
      <bpmn:outgoing>Flow_1</bpmn:outgoing>
    </bpmn:startEvent>

    <!-- Sequence Flow 1 -->
    <bpmn:sequenceFlow id="Flow_1" sourceRef="StartEvent_1" targetRef="Task_Validate"/>

    <!-- Task 1: Validate Customer Info -->
    <bpmn:serviceTask id="Task_Validate" name="Validate Customer Information">
      <bpmn:incoming>Flow_1</bpmn:incoming>
      <bpmn:outgoing>Flow_2</bpmn:outgoing>
    </bpmn:serviceTask>

    <!-- Sequence Flow 2 -->
    <bpmn:sequenceFlow id="Flow_2" sourceRef="Task_Validate" targetRef="Task_CheckCredit"/>

    <!-- Task 2: Check Credit Score -->
    <bpmn:serviceTask id="Task_CheckCredit" name="Check Credit Score">
      <bpmn:incoming>Flow_2</bpmn:incoming>
      <bpmn:outgoing>Flow_3</bpmn:outgoing>
    </bpmn:serviceTask>

    <!-- Sequence Flow 3 -->
    <bpmn:sequenceFlow id="Flow_3" sourceRef="Task_CheckCredit" targetRef="Gateway_CreditScore"/>

    <!-- Exclusive Gateway: Decision based on credit score -->
    <bpmn:exclusiveGateway id="Gateway_CreditScore" name="Credit Score >= 700?">
      <bpmn:incoming>Flow_3</bpmn:incoming>
      <bpmn:outgoing>Flow_Approve</bpmn:outgoing>
      <bpmn:outgoing>Flow_ManualReview</bpmn:outgoing>
      <bpmn:outgoing>Flow_Reject</bpmn:outgoing>
    </bpmn:exclusiveGateway>

    <!-- Flow: Auto Approve (credit score >= 700) -->
    <bpmn:sequenceFlow id="Flow_Approve" sourceRef="Gateway_CreditScore" targetRef="Task_AutoApprove">
      <bpmn:conditionExpression>${creditScore >= 700}</bpmn:conditionExpression>
    </bpmn:sequenceFlow>

    <bpmn:serviceTask id="Task_AutoApprove" name="Auto Approve Loan">
      <bpmn:incoming>Flow_Approve</bpmn:incoming>
      <bpmn:outgoing>Flow_ApproveToNotify</bpmn:outgoing>
    </bpmn:serviceTask>

    <bpmn:sequenceFlow id="Flow_ApproveToNotify" sourceRef="Task_AutoApprove" targetRef="Task_Notify"/>

    <!-- Flow: Manual Review (500 <= credit score < 700) -->
    <bpmn:sequenceFlow id="Flow_ManualReview" sourceRef="Gateway_CreditScore" targetRef="Task_ManualReview">
      <bpmn:conditionExpression>${creditScore >= 500 && creditScore < 700}</bpmn:conditionExpression>
    </bpmn:sequenceFlow>

    <bpmn:userTask id="Task_ManualReview" name="Manual Review by Employee">
      <bpmn:incoming>Flow_ManualReview</bpmn:incoming>
      <bpmn:outgoing>Flow_ManualToNotify</bpmn:outgoing>
    </bpmn:userTask>

    <bpmn:sequenceFlow id="Flow_ManualToNotify" sourceRef="Task_ManualReview" targetRef="Task_Notify"/>

    <!-- Flow: Reject (credit score < 500) -->
    <bpmn:sequenceFlow id="Flow_Reject" sourceRef="Gateway_CreditScore" targetRef="Task_Reject">
      <bpmn:conditionExpression>${creditScore < 500}</bpmn:conditionExpression>
    </bpmn:sequenceFlow>

    <bpmn:serviceTask id="Task_Reject" name="Reject Loan Application">
      <bpmn:incoming>Flow_Reject</bpmn:incoming>
      <bpmn:outgoing>Flow_RejectToNotify</bpmn:outgoing>
    </bpmn:serviceTask>

    <bpmn:sequenceFlow id="Flow_RejectToNotify" sourceRef="Task_Reject" targetRef="Task_Notify"/>

    <!-- Task: Send Notification -->
    <bpmn:serviceTask id="Task_Notify" name="Send Notification to Customer">
      <bpmn:incoming>Flow_ApproveToNotify</bpmn:incoming>
      <bpmn:incoming>Flow_ManualToNotify</bpmn:incoming>
      <bpmn:incoming>Flow_RejectToNotify</bpmn:incoming>
      <bpmn:outgoing>Flow_4</bpmn:outgoing>
    </bpmn:serviceTask>

    <!-- Sequence Flow 4 -->
    <bpmn:sequenceFlow id="Flow_4" sourceRef="Task_Notify" targetRef="EndEvent_1"/>

    <!-- End Event -->
    <bpmn:endEvent id="EndEvent_1" name="Process Completed">
      <bpmn:incoming>Flow_4</bpmn:incoming>
    </bpmn:endEvent>

  </bpmn:process>

  <!-- BPMN Diagram (Visual positioning) -->
  <bpmndi:BPMNDiagram id="BPMNDiagram_1">
    <bpmndi:BPMNPlane id="BPMNPlane_1" bpmnElement="loanApproval">
      <!-- Shapes and positions would be defined here -->
    </bpmndi:BPMNPlane>
  </bpmndi:BPMNDiagram>

</bpmn:definitions>
```

### b. Java Delegate cho các Tasks

```java
// ValidateCustomerDelegate.java
@Component
public class ValidateCustomerDelegate implements JavaDelegate {
    
    @Override
    public void execute(DelegateExecution execution) {
        String applicationId = (String) execution.getVariable("applicationId");
        
        // Validate customer info from database
        Customer customer = customerService.findById(applicationId);
        
        if (customer == null || !customer.isValid()) {
            throw new IllegalArgumentException("Invalid customer");
        }
        
        execution.setVariable("customerValidated", true);
        System.out.println("Customer validated: " + customer.getName());
    }
}
```

### d. Start Process từ Application

```java
// LoanApplicationService.java
@Service
public class LoanApplicationService {
    
    @Autowired
    private RuntimeService runtimeService;
    
    public LoanApplicationResponse applyForLoan(LoanApplicationRequest request) {
        // Start BPMN process
        ProcessInstance processInstance = runtimeService.startProcessInstanceByKey(
            "loanApproval",
            Variables.createVariables()
                .putValue("applicationId", request.getApplicationId())
                .putValue("customerId", request.getCustomerId())
                .putValue("loanAmount", request.getAmount())
                .putValue("loanPurpose", request.getPurpose())
        );
        
        return new LoanApplicationResponse(
            processInstance.getId(),
            processInstance.getProcessDefinitionId(),
            "PENDING"
        );
    }
}
```

### b. Temporal với TypeScript (Code-only)

```typescript
// loan-approval.workflow.ts
import { proxyActivities, sleep } from '@temporalio/workflow';
import type * as activities from './activities';

const { 
  validateCustomer, 
  checkCreditScore, 
  approveLoan, 
  rejectLoan, 
  sendNotification 
} = proxyActivities<typeof activities>({
  startToCloseTimeout: '1 minute',
  retry: { initialInterval: '1s', maxAttempts: 3 }
});

export async function loanApprovalWorkflow(
  applicationId: string,
  customerId: string,
  loanAmount: number
): Promise<LoanDecision> {
  
  // Step 1: Validate customer
  await validateCustomer(customerId);
  
  // Step 2: Check credit score
  const creditScore = await checkCreditScore(customerId);
  
  // Step 3: Decision (Exclusive Gateway)
  let decision: LoanDecision;
  
  if (creditScore >= 700) {
    // Auto approve
    await approveLoan(applicationId, loanAmount, 'AUTO');
    decision = { status: 'APPROVED', type: 'AUTO' };
    
  } else if (creditScore >= 500) {
    // Manual review - wait for human decision
    const manualDecision = await waitForManualReview(applicationId);
    
    if (manualDecision.approved) {
      await approveLoan(applicationId, loanAmount, 'MANUAL');
      decision = { status: 'APPROVED', type: 'MANUAL' };
    } else {
      await rejectLoan(applicationId, manualDecision.reason);
      decision = { status: 'REJECTED', reason: manualDecision.reason };
    }
    
  } else {
    // Auto reject
    await rejectLoan(applicationId, 'Credit score too low');
    decision = { status: 'REJECTED', reason: 'Credit score too low' };
  }
  
  // Step 4: Send notification
  await sendNotification(customerId, decision);
  
  return decision;
}

// Helper function for manual review
export async function waitForManualReview(applicationId: string): Promise<ManualDecision> {
  // Wait for external signal (human decision)
  const signal = await workflow.signal<ManualDecision>('manualReviewDecision');
  return signal;
}
```

```typescript
// activities.ts
import { activity } from '@temporalio/activity';

export async function validateCustomer(customerId: string): Promise<void> {
  // Call customer service API
  const response = await fetch(`/api/customers/${customerId}`);
  if (!response.ok) throw new Error('Customer not found');
}

export async function checkCreditScore(customerId: string): Promise<number> {
  // Call credit bureau API
  const response = await fetch(`/api/credit-score/${customerId}`);
  const data = await response.json();
  return data.score;
}

export async function approveLoan(
  applicationId: string, 
  amount: number,
  approvalType: string
): Promise<void> {
  // Call loan service API
  await fetch(`/api/loans/${applicationId}/approve`, {
    method: 'POST',
    body: JSON.stringify({ amount, approvalType })
  });
}

export async function rejectLoan(
  applicationId: string, 
  reason: string
): Promise<void> {
  await fetch(`/api/loans/${applicationId}/reject`, {
    method: 'POST',
    body: JSON.stringify({ reason })
  });
}

export async function sendNotification(
  customerId: string, 
  decision: LoanDecision
): Promise<void> {
  await fetch('/api/notifications/send', {
    method: 'POST',
    body: JSON.stringify({ customerId, decision })
  });
}
```

## 7. Bảng so sánh trực quan

| Khía cạnh | BPMN (Camunda) | Temporal |
|-----------|----------------|----------|
| **Định nghĩa** | XML file (kéo thả visual) | TypeScript/Go/Java code |
| **Học curve** | Dễ (visual, kéo thả) | Trung bình (cần code) |
| **Business collaboration** | ✅ Tốt (business hiểu được) | ❌ Chỉ developers |
| **Version control** | XML diff (khó đọc) | Git code diff (dễ đọc) |
| **Testing** | Integration test | Unit test + Integration test |
| **Debugging** | Camunda Cockpit UI | Temporal UI + Debugger |
| **Flexibility** | Hạn chế (theo BPMN chuẩn) | Cao (code làm gì cũng được) |
| **Documentation** | ✅ Tự động từ diagram | ❌ Phải tự viết docs |

## 8. Kết luận

### BPMN phù hợp khi:

```
✅ Bạn cần:
├── Visual diagram cho business stakeholders
├── Low-code configuration
├── Compliance documentation
├── Process mining và analysis
└── Collaboration giữa business và IT
```

```
✅ Bạn cần:

├── Unit test coverage cao
├── Complex business logic trong code
├── Distributed system orchestration
└── Developer productivity
```
