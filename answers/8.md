# Camunda (XML) vs Temporal (Code) - Cách Định Nghĩa Workflow

## 1. Câu trả lời ngắn gọn

**ĐÚNG!** 

| Tool | Cách định nghĩa | File |
|------|----------------|------|
| **Camunda** | ✅ BPMN 2.0 XML (visual diagram) | `.bpmn` file |
| **Temporal** | ✅ Code (SDK) | `.ts`, `.java`, `.go`, `.py` file |

## 2. Camunda - XML-based (BPMN 2.0)

### a. Ví dụ BPMN XML

```xml
<?xml version="1.0" encoding="UTF-8"?>
<bpmn:definitions xmlns:bpmn="http://www.omg.org/spec/BPMN/20100524/MODEL"
                  xmlns:camunda="http://camunda.org/schema/1.0/bpmn">

  <bpmn:process id="orderFulfillment" name="Order Fulfillment" isExecutable="true">
    
    <!-- Start Event -->
    <bpmn:startEvent id="start" name="Order Received">
      <bpmn:outgoing>flow1</bpmn:outgoing>
    </bpmn:startEvent>
    
    <!-- Sequence Flow -->
    <bpmn:sequenceFlow id="flow1" sourceRef="start" targetRef="validateOrder"/>
    
    <!-- Service Task 1 -->
    <bpmn:serviceTask id="validateOrder" name="Validate Order">
      <bpmn:extensionElements>
        <camunda:delegateExpression>${validateOrderDelegate}</camunda:delegateExpression>
      </bpmn:extensionElements>
      <bpmn:incoming>flow1</bpmn:incoming>
      <bpmn:outgoing>flow2</bpmn:outgoing>
    </bpmn:serviceTask>
    
    <!-- Sequence Flow -->
    <bpmn:sequenceFlow id="flow2" sourceRef="validateOrder" targetRef="checkInventory"/>
    
    <!-- Service Task 2 -->
    <bpmn:serviceTask id="checkInventory" name="Check Inventory">
      <bpmn:extensionElements>
        <camunda:delegateExpression>${checkInventoryDelegate}</camunda:delegateExpression>
      </bpmn:extensionElements>
      <bpmn:incoming>flow2</bpmn:incoming>
      <bpmn:outgoing>flow3</bpmn:outgoing>
    </bpmn:serviceTask>
    
    <!-- Sequence Flow -->
    <bpmn:sequenceFlow id="flow3" sourceRef="checkInventory" targetRef="gateway"/>
    
    <!-- Exclusive Gateway (Decision) -->
    <bpmn:exclusiveGateway id="gateway" name="In Stock?">
      <bpmn:incoming>flow3</bpmn:incoming>
      <bpmn:outgoing>flow4</bpmn:outgoing>
      <bpmn:outgoing>flow5</bpmn:outgoing>
    </bpmn:exclusiveGateway>
    
    <!-- Flow: Yes (in stock) -->
    <bpmn:sequenceFlow id="flow4" sourceRef="gateway" targetRef="shipOrder">
      <bpmn:conditionExpression>${inStock == true}</bpmn:conditionExpression>
    </bpmn:sequenceFlow>
    
    <bpmn:serviceTask id="shipOrder" name="Ship Order">
      <bpmn:extensionElements>
        <camunda:delegateExpression>${shipOrderDelegate}</camunda:delegateExpression>
      </bpmn:extensionElements>
    </bpmn:serviceTask>
    
    <!-- Flow: No (out of stock) -->
    <bpmn:sequenceFlow id="flow5" sourceRef="gateway" targetRef="notifyBackorder">
      <bpmn:conditionExpression>${inStock == false}</bpmn:conditionExpression>
    </bpmn:sequenceFlow>
    
    <bpmn:serviceTask id="notifyBackorder" name="Notify Backorder">
      <bpmn:extensionElements>
        <camunda:delegateExpression>${notifyBackorderDelegate}</camunda:delegateExpression>
      </bpmn:extensionElements>
    </bpmn:serviceTask>
    
    <!-- End Event -->
    <bpmn:endEvent id="end"/>
    
  </bpmn:process>
</bpmn:definitions>
```

## 3. Temporal - Code-based

### a. Ví dụ TypeScript Workflow

```typescript
// order-fulfillment.workflow.ts
import { proxyActivities, sleep } from '@temporalio/workflow';
import type * as activities from './activities';

const { 
  validateOrder, 
  checkInventory, 
  shipOrder, 
  notifyBackorder 
} = proxyActivities<typeof activities>({
  startToCloseTimeout: '1 minute',
  retry: { 
    initialInterval: '1s', 
    maxAttempts: 3 
  }
});

export async function orderFulfillmentWorkflow(
  orderId: string,
  customerId: string,
  items: OrderItem[]
): Promise<WorkflowResult> {
  
  // Step 1: Validate order
  const isValid = await validateOrder(orderId, customerId, items);
  if (!isValid) {
    throw new ApplicationError('Invalid order', orderId);
  }
  
  // Step 2: Check inventory
  const inventoryStatus = await checkInventory(items);
  
  // Step 3: Decision (Exclusive Gateway equivalent)
  if (inventoryStatus.inStock) {
    // In stock - ship immediately
    await shipOrder(orderId, items);
    return { status: 'SHIPPED', orderId };
    
  } else {
    // Out of stock - notify backorder
    await notifyBackorder(orderId, items, inventoryStatus.estimatedRestockDate);
    return { status: 'BACKORDER', orderId, estimatedDate: inventoryStatus.estimatedRestockDate };
  }
}
```

### b. Activities Implementation

```typescript
// activities.ts
import { activity } from '@temporalio/activity';

export async function validateOrder(
  orderId: string, 
  customerId: string, 
  items: OrderItem[]
): Promise<boolean> {
  // Call order service API
  const response = await fetch(`/api/orders/${orderId}/validate`, {
    method: 'POST',
    body: JSON.stringify({ orderId, customerId, items })
  });
  
  if (!response.ok) {
    throw new ApplicationError('Order validation failed');
  }
  
  const data = await response.json();
  return data.valid;
}

export async function checkInventory(items: OrderItem[]): Promise<InventoryStatus> {
  // Call inventory service API
  const response = await fetch('/api/inventory/check', {
    method: 'POST',
    body: JSON.stringify({ items })
  });
  
  const data = await response.json();
  return {
    inStock: data.inStock,
    estimatedRestockDate: data.estimatedRestockDate
  };
}

export async function shipOrder(orderId: string, items: OrderItem[]): Promise<void> {
  // Call shipping service API
  await fetch('/api/shipping/ship', {
    method: 'POST',
    body: JSON.stringify({ orderId, items })
  });
}

export async function notifyBackorder(
  orderId: string, 
  items: OrderItem[], 
  estimatedDate: Date
): Promise<void> {
  // Call notification service API
  await fetch('/api/notifications/backorder', {
    method: 'POST',
    body: JSON.stringify({ orderId, items, estimatedDate })
  });
}
```

## 4. So sánh trực quan

### a. Cách thể hiện cùng một quy trình

| Khía cạnh | Camunda (BPMN) | Temporal (Code) |
|-----------|----------------|-----------------|
| **Định nghĩa** | XML file | TypeScript/Java/Go/Python code |
| **Flow control** | Sequence flows, gateways | if/else, for/while |
| **Decision** | Exclusive Gateway (◇) | if/else statement |
| **Parallel** | Parallel Gateway (◇ with +) | Promise.all() |
| **Timer** | Timer Event (🕐) | sleep() function |
| **Error handling** | Error Boundary Event | try/catch block |
| **Retry** | Retry configuration in XML | Retry options in proxy |

### b. Visual so sánh

## 5. Bảng so sánh chi tiết

| Tiêu chí | Camunda (XML) | Temporal (Code) |
|----------|---------------|-----------------|
| **File format** | `.bpmn` (XML) | `.ts`, `.java`, `.go`, `.py` |
| **Tool để tạo** | Camunda Modeler (drag & drop) | VS Code, IntelliJ, etc. |
| **Learning curve** | Thấp (visual, dễ học) | Trung bình (cần biết code) |
| **Business collaboration** | ✅ Tốt (BA hiểu được) | ❌ Chỉ developers |
| **Version control** | XML diff (khó đọc) | Code diff (dễ đọc với Git) |
| **Testing** | Integration test | Unit test + Integration test |
| **Debugging** | Camunda Cockpit UI | Debugger + Temporal UI |
| **Refactoring** | Khó (phải sửa XML) | Dễ (IDE support) |
| **Documentation** | ✅ Tự động từ diagram | ❌ Phải tự viết docs |
| **Flexibility** | Hạn chế (theo BPMN chuẩn) | Cao (code làm gì cũng được) |
| **Deployment** | Deploy BPMN file + JAR | Deploy code bundle |

## 7. Khi nào chọn cái nào?

### Chọn Camunda (XML/BPMN) khi:

```
✅ Bạn cần:
├── Visual diagram cho business stakeholders
├── Business analysts thiết kế workflow
├── Compliance yêu cầu process documentation
├── Low-code configuration
├── Process hay thay đổi bởi business users
└── Cần BPMN 2.0 standard

Ví dụ:
├── Loan approval (ngân hàng)
├── Insurance claims (bảo hiểm)
├── Order fulfillment (e-commerce)
├── HR onboarding
└── Customer complaint handling
```

### Chọn Temporal (Code) khi:

```
✅ Bạn cần:
├── Code-first approach
├── Unit test coverage cao
├── Complex business logic trong workflow
├── Distributed system orchestration
├── Developer productivity
├── Version control với Git
└── Microservices coordination

Ví dụ:
├── Data pipeline orchestration
├── Microservices coordination
├── Event-driven architectures
├── Distributed transaction management
├── Real-time processing workflows
└── CI/CD pipeline orchestration
```

## 8. Ví dụ cùng 1 quy trình - 2 cách implement

### Camunda (BPMN XML + Java)

```xml
<!-- order-process.bpmn -->
<bpmn:process id="orderProcess" isExecutable="true">
  <bpmn:startEvent id="start"/>
  <bpmn:serviceTask id="validate" camunda:delegateExpression="${validateDelegate}"/>
  <bpmn:exclusiveGateway id="gateway"/>
  <bpmn:serviceTask id="ship" camunda:delegateExpression="${shipDelegate}"/>
  <bpmn:serviceTask id="backorder" camunda:delegateExpression="${backorderDelegate}"/>
  <bpmn:endEvent id="end"/>
  
  <bpmn:sequenceFlow sourceRef="start" targetRef="validate"/>
  <bpmn:sequenceFlow sourceRef="validate" targetRef="gateway"/>
  <bpmn:sequenceFlow sourceRef="gateway" targetRef="ship">
    <bpmn:conditionExpression>${inStock}</bpmn:conditionExpression>
  </bpmn:sequenceFlow>
  <bpmn:sequenceFlow sourceRef="gateway" targetRef="backorder">
    <bpmn:conditionExpression>${!inStock}</bpmn:conditionExpression>
  </bpmn:sequenceFlow>
  <bpmn:sequenceFlow sourceRef="ship" targetRef="end"/>
  <bpmn:sequenceFlow sourceRef="backorder" targetRef="end"/>
</bpmn:process>
```


### Temporal (TypeScript only)

```typescript
// order-workflow.ts
export async function orderWorkflow(order: Order) {
  await validate(order);
  
  const inventory = await checkInventory(order.items);
  
  if (inventory.inStock) {
    await ship(order);
    return { status: 'SHIPPED' };
  } else {
    await backorder(order, inventory.restockDate);
    return { status: 'BACKORDER' };
  }
}
```
